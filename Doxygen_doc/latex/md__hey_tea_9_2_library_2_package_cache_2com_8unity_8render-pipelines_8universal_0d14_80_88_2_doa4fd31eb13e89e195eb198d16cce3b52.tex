 This page describes how to upgrade from an older version of the Universal Render Pipeline (URP) to version 9.\+0.\+x.\hypertarget{md__hey_tea_9_2_library_2_package_cache_2com_8unity_8render-pipelines_8universal_0d14_80_88_2_doa4fd31eb13e89e195eb198d16cce3b52_autotoc_md2619}{}\doxysection{\texorpdfstring{Upgrading from URP 8.\+0.\+x and later 8.\+x releases}{Upgrading from URP 8.\+0.\+x and later 8.\+x releases}}\label{md__hey_tea_9_2_library_2_package_cache_2com_8unity_8render-pipelines_8universal_0d14_80_88_2_doa4fd31eb13e89e195eb198d16cce3b52_autotoc_md2619}

\begin{DoxyEnumerate}
\item URP 9.\+0.\+x does not have breaking changes compared with URP 8.\+x.\+x. To upgrade URP to version 9.\+0.\+x, install the new version of the package.
\end{DoxyEnumerate}\hypertarget{md__hey_tea_9_2_library_2_package_cache_2com_8unity_8render-pipelines_8universal_0d14_80_88_2_doa4fd31eb13e89e195eb198d16cce3b52_autotoc_md2620}{}\doxysection{\texorpdfstring{Upgrading from URP 7.\+2.\+x and later 7.\+x releases}{Upgrading from URP 7.\+2.\+x and later 7.\+x releases}}\label{md__hey_tea_9_2_library_2_package_cache_2com_8unity_8render-pipelines_8universal_0d14_80_88_2_doa4fd31eb13e89e195eb198d16cce3b52_autotoc_md2620}

\begin{DoxyEnumerate}
\item URP 9.\+x.\+x does not support the package Post-\/\+Processing Stack v2. If your Project uses the package Post-\/\+Processing Stack v2, migrate the effects that use that package first.
\end{DoxyEnumerate}\hypertarget{md__hey_tea_9_2_library_2_package_cache_2com_8unity_8render-pipelines_8universal_0d14_80_88_2_doa4fd31eb13e89e195eb198d16cce3b52_autotoc_md2621}{}\doxysection{\texorpdfstring{Upgrading from URP 7.\+0.\+x-\/7.\+1.\+x}{Upgrading from URP 7.\+0.\+x-\/7.\+1.\+x}}\label{md__hey_tea_9_2_library_2_package_cache_2com_8unity_8render-pipelines_8universal_0d14_80_88_2_doa4fd31eb13e89e195eb198d16cce3b52_autotoc_md2621}

\begin{DoxyEnumerate}
\item Upgrade to URP 7.\+2.\+0 first. Refer to \doxysectlink{md__hey_tea_9_2_library_2_package_cache_2com_8unity_8render-pipelines_8universal_0d14_80_88_2_dod7feaf10650c864ca40427e6ba472f17}{Upgrading to version 7.2.0 of the Universal Render Pipeline}{0}.
\item URP 8.\+x.\+x does not support the package Post-\/\+Processing Stack v2. If your Project uses the package Post-\/\+Processing Stack v2, migrate the effects that use that package first. 
\end{DoxyEnumerate}
